Sum Of All Fears

  • A challenge to demonstrate searching through arrays in JavaScript
  • Enter in a Number 'K' and click the button, the program will then search through an array of numbers and determine if the sum of any two number in the array will add up to your number.

document.getElementById("click").addEventListener("click", () => {
	let kNum = parseInt(document.getElementById("kNum").value);
								
	let arr = new Array;
	arr = [10, 15, 3, 7];
	if (checkNum(arr, kNum) === true) {
		document.getElementById("result").innerHTML = "your number, " + kNum +
						", does have numbers in the array that equal it.";
	} else {
		document.getElementById("result").innerHTML = "your number, " + kNum + 
						", does not have numbers in the array that equal it.";
	}
})
function checkNum(arr, kNum) {
	let trueVal = -1
	for (let i = 0; i < arr.length; i++) {
		trueVal = arr.indexOf(kNum - arr[i]);
	}
	if (trueVal != -1) {
		return true;
	} else {
		return false;
	}
}
